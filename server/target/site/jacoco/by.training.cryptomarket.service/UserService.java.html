    /**
     *
     * This method checks the presence of the user.
     * @param name name
     * @return boolean value
     * @throws Exception Exception
     */
    public boolean userIsExist(final String name) throws Exception {
         List<User> users;


         users = userDao.read();

        for (User user : users) {

            if (name.equals(user.getUserName())) {
                return true;
            }
        }
        return false;
    }
